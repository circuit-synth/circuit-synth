from circuit_synth import *

# Define a professional ESP32 circuit with Python
@circuit(name="esp32c6_simple")
def esp32c6_simple():
    """Simple ESP32-C6 circuit with decoupling capacitor"""
    
    # Create power nets
    _3V3 = Net('3V3')
    GND = Net('GND')
    
    # ESP32-C6 module
    esp32c6 = Component(
        symbol="RF_Module:ESP32-C6-MINI-1",
        ref="U",
        footprint="RF_Module:ESP32-C6-MINI-1"
    )
    
    # Power connections
    esp32c6["3V3"] += _3V3
    esp32c6["GND"] += GND

# Generate professional KiCad project
circuit = esp32c6_simple()
circuit.generate_kicad_project("esp32c6_simple")

Get Started in Minutes

Try Circuit Synth immediately without installation, or set up your development environment

1

Quick Trial

Test Circuit Synth immediately without any setup:

git clone https://github.com/circuit-synth/circuit-synth.git cd circuit-synth uv run python examples/example_kicad_project.py
2

Install for Development

Set up your local development environment:

curl -LsSf https://astral.sh/uv/install.sh | sh uv pip install -e ".[dev]"
3

Run with Docker

Use the pre-configured Docker environment:

./docker/build-docker.sh ./scripts/circuit-synth-docker python examples/example_kicad_project.py
